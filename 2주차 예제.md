## 조건문
### Java

```
public class ControlFlowExample {
    public static void main(String[] args) {
        // 1. if-else
        int score = 85;
        if (score >= 90) {
            System.out.println("A등급");
        } else if (score >= 80) {
            System.out.println("B등급");
        } else {
            System.out.println("C등급");
        }

        // 2. switch
        String grade = "A";
        switch (grade) {
            case "A":
                System.out.println("우수");
                break;
            case "B":
                System.out.println("양호");
                break;
            default:
                System.out.println("보통");
        }


    }
}
```
### kotlin
```
fun main() {

    val number = 5

        in 1..10 -> println("1에서 10 사이")
        in 11..20 -> println("11에서 20 사이")

    }
}

##반복문

### Java
```
public class LoopExample {
    public static void main(String[] args) {
        // 1. for 루프
        for (int i = 0; i < 5; i++) {
            System.out.println("Count: " + i);
        }

        // 2. while 루프
        int count = 0;
        while (count < 3) {
            System.out.println("While count: " + count);
            count++;
        }

        // 3. do-while 루프
        int num = 0;
        do {
            System.out.println("Do-while num: " + num);
            num++;
        } while (num < 2);

        // 4. for-each 루프
        String[] fruits = {"사과", "바나나", "오렌지"};
        for (String fruit : fruits) {
            System.out.println(fruit);
        }

        // 5. break와 continue
        for (int i = 0; i < 5; i++) {
            if (i == 2) continue; // 2를 건너뜀
            if (i == 4) break;    // 4에서 종료
            System.out.println("Value: " + i);
        }
    }
}
```

### kotlin
```
fun main() {
    // 1. for 루프
    // 범위를 사용한 반복
    for (i in 0..4) {  // 0부터 4까지 (4 포함)
        println("Count: $i")
    }

    // until 사용 (마지막 숫자 제외)
    for (i in 0 until 5) {  // 0부터 4까지
        println("Until count: $i")
    }

    // step 사용
    for (i in 0..10 step 2) {  // 2씩 증가
        println("Step count: $i")
    }

    // 감소하는 범위
    for (i in 5 downTo 1) {
        println("Countdown: $i")
    }

    // 2. while 루프
    var count = 0
    while (count < 3) {
        println("While count: $count")
        count++
    }

    // 3. do-while 루프
    var num = 0
    do {
        println("Do-while num: $num")
        num++
    } while (num < 2)

    // 4. 컬렉션 반복
    val fruits = arrayOf("사과", "바나나", "오렌지")
    for (fruit in fruits) {
        println(fruit)
    }

    // indices를 사용한 인덱스 접근
    for (index in fruits.indices) {
        println("${index}번째 과일: ${fruits[index]}")
    }

    // withIndex()를 사용한 인덱스와 값 동시 접근
    for ((index, value) in fruits.withIndex()) {
        println("$index: $value")
    }

    // 5. break와 continue 사용
    for (i in 0..5) {
        if (i == 2) continue  // 2를 건너뜀
        if (i == 4) break     // 4에서 종료
        println("Value: $i")
    }
}

## null 안정성

### java
```
public class NullSafetyExample {
    public static void main(String[] args) {
        // Java에서의 null 처리
        String name = null;
        
        // null 체크가 필요
        if (name != null) {
            System.out.println(name.length());
        }
        
        // Optional 사용 (Java 8 이상)
        Optional<String> optionalName = Optional.ofNullable(name);
        optionalName.ifPresent(n -> System.out.println(n.length()));
        
        // null 체크를 하지 않으면 NullPointerException 발생 가능
        try {
            System.out.println(name.length()); // NullPointerException!
        } catch (NullPointerException e) {
            System.out.println("Null 포인터 예외 발생!");
        }
    }
    
    // null을 반환할 수 있는 메서드
    public static String getNullableName() {
        return null;
    }
}

```

### kotlin
```
fun main() {
    // Nullable과 Non-null 변수 선언
    var nullableStr: String? = null
    var nonNullStr: String = "Hello"

    // 1. ? (Nullable 타입 선언)
    var name: String? = null  // null 허용
    // var name2: String = null  // 컴파일 에러: null 불가능

    // 2. ?. (안전 호출 연산자)
    println(nullableStr?.length)  // null 출력
    println(nonNullStr.length)    // 5 출력

    // 3. !! (null 아님 단언)
    try {
        println(nullableStr!!.length)  // NullPointerException 발생
    } catch (e: NullPointerException) {
        println("NPE 발생!")
    }

    // 4. ?. 체이닝 예제
    val str: String? = "Kotlin"
    println(str?.uppercase()?.length)  // 6 출력
    
    str = null
    println(str?.uppercase()?.length)  // null 출력
}
```

public class ArrayExample {
    public static void main(String[] args) {
        // 1. 배열 선언과 초기화
        int[] numbers = {1, 2, 3, 4, 5};  // 선언과 동시에 초기화
        String[] fruits = new String[3];   // 크기만 지정하여 선언
        
        // 2. 배열 요소 접근과 수정
        fruits[0] = "apple";
        fruits[1] = "banana";
        fruits[2] = "orange";
        
        System.out.println(numbers[0]);    // 1
        System.out.println(fruits[0]);     // apple
        
        // 3. 배열 길이
        System.out.println("배열 길이: " + numbers.length);
        
        // 4. 배열 순회
        // 기본 for문
        for (int i = 0; i < numbers.length; i++) {
            System.out.println(numbers[i]);
        }
        
        // for-each문
        for (String fruit : fruits) {
            System.out.println(fruit);
        }
        
        // 5. 2차원 배열
        int[][] matrix = new int[3][3];  // 3x3 배열
        matrix[0][0] = 1;
        
        // 2차원 배열 초기화
        int[][] matrix2 = {
            {1, 2, 3},
            {4, 5, 6},
            {7, 8, 9}
        };
        
        // 6. 배열 복사
        int[] copyNumbers = new int[numbers.length];
        System.arraycopy(numbers, 0, copyNumbers, 0, numbers.length);
        
        // Arrays 클래스 사용
        int[] copyNumbers2 = Arrays.copyOf(numbers, numbers.length);
        
        // 7. 배열 정렬
        int[] unsorted = {3, 1, 4, 1, 5};
        Arrays.sort(unsorted);
        System.out.println(Arrays.toString(unsorted));  // [1, 1, 3, 4, 5]
        
        // 8. 배열 검색
        int index = Arrays.binarySearch(unsorted, 3);  // 정렬된 배열에서 검색
        
        // 9. 배열을 리스트로 변환
        List<String> fruitsList = Arrays.asList(fruits);
        
        // 10. 배열 채우기
        int[] filled = new int[5];
        Arrays.fill(filled, 7);  // [7, 7, 7, 7, 7]
    }
}

fun main() {
