### Java

```
public class ControlFlowExample {
    public static void main(String[] args) {
        // 1. if-else
        int score = 85;
        if (score >= 90) {
            System.out.println("A등급");
        } else if (score >= 80) {
            System.out.println("B등급");
        } else {
            System.out.println("C등급");
        }

        // 2. switch
        String grade = "A";
        switch (grade) {
            case "A":
                System.out.println("우수");
                break;
            case "B":
                System.out.println("양호");
                break;
            default:
                System.out.println("보통");
        }

        // 3. 삼항 연산자
        int age = 20;
        String status = (age >= 18) ? "성인" : "미성년자";
        System.out.println(status);
    }
}
```
### kotlin
```
fun main() {
    // 1. if-else (Kotlin에서는 표현식으로 사용 가능)
    val score = 85
    val grade = if (score >= 90) {
        "A등급"
    } else if (score >= 80) {
        "B등급"
    } else {
        "C등급"
    }
    println(grade)

    // 2. when (Java의 switch와 유사하지만 더 강력)
    val gradeLevel = "A"
    when (gradeLevel) {
        "A" -> println("우수")
        "B" -> println("양호")
        else -> println("보통")
    }

    // when을 표현식으로 사용
    val result = when (gradeLevel) {
        "A", "B" -> "합격"
        else -> "재시험"
    }
    println(result)

    // 3. 범위 체크
    val number = 5
    when (number) {
        in 1..10 -> println("1에서 10 사이")
        in 11..20 -> println("11에서 20 사이")
        else -> println("범위 밖")
    }
}
```

### Java
```
public class LoopExample {
    public static void main(String[] args) {
        // 1. for 루프
        for (int i = 0; i < 5; i++) {
            System.out.println("Count: " + i);
        }

        // 2. while 루프
        int count = 0;
        while (count < 3) {
            System.out.println("While count: " + count);
            count++;
        }

        // 3. do-while 루프
        int num = 0;
        do {
            System.out.println("Do-while num: " + num);
            num++;
        } while (num < 2);

        // 4. for-each 루프
        String[] fruits = {"사과", "바나나", "오렌지"};
        for (String fruit : fruits) {
            System.out.println(fruit);
        }

        // 5. break와 continue
        for (int i = 0; i < 5; i++) {
            if (i == 2) continue; // 2를 건너뜀
            if (i == 4) break;    // 4에서 종료
            System.out.println("Value: " + i);
        }
    }
}
```

### kotlin
```
fun main() {
    // 1. for 루프
    // 범위를 사용한 반복
    for (i in 0..4) {  // 0부터 4까지 (4 포함)
        println("Count: $i")
    }

    // until 사용 (마지막 숫자 제외)
    for (i in 0 until 5) {  // 0부터 4까지
        println("Until count: $i")
    }

    // step 사용
    for (i in 0..10 step 2) {  // 2씩 증가
        println("Step count: $i")
    }

    // 감소하는 범위
    for (i in 5 downTo 1) {
        println("Countdown: $i")
    }

    // 2. while 루프
    var count = 0
    while (count < 3) {
        println("While count: $count")
        count++
    }

    // 3. do-while 루프
    var num = 0
    do {
        println("Do-while num: $num")
        num++
    } while (num < 2)

    // 4. 컬렉션 반복
    val fruits = arrayOf("사과", "바나나", "오렌지")
    for (fruit in fruits) {
        println(fruit)
    }

    // indices를 사용한 인덱스 접근
    for (index in fruits.indices) {
        println("${index}번째 과일: ${fruits[index]}")
    }

    // withIndex()를 사용한 인덱스와 값 동시 접근
    for ((index, value) in fruits.withIndex()) {
        println("$index: $value")
    }

    // 5. break와 continue 사용
    for (i in 0..5) {
        if (i == 2) continue  // 2를 건너뜀
        if (i == 4) break     // 4에서 종료
        println("Value: $i")
    }
}
```
